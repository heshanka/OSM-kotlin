77      @Test
78      public void verifyMessageFormatting() {
79          Integer i1 = new Integer(1);
80          Integer i2 = new Integer(2);
81          Integer i3 = new Integer(3);
82          Exception e = new Exception("This is a test exception.");
83          Logger logger = LoggerFactory.getLogger("test2");
84  
85          int index = 0;
86          logger.debug("Hello world");
87          assertLogMessage("Hello world", index++);
88  
89          logger.debug("Hello world {}", i1);
90          assertLogMessage("Hello world " + i1, index++);
91  
92          logger.debug("val={} val={}", i1, i2);
93          assertLogMessage("val=1 val=2", index++);
94  
95          logger.debug("val={} val={} val={}", new Object[] { i1, i2, i3 });
96          assertLogMessage("val=1 val=2 val=3", index++);
97  
98          logger.debug("Hello world 2", e);
99          assertLogMessage("Hello world 2", index);
100         assertException(e.getClass(), index++);
101         logger.info("Hello world 2.");
102 
103         logger.warn("Hello world 3.");
104         logger.warn("Hello world 3", e);
105 
106         logger.error("Hello world 4.");
107         logger.error("Hello world {}", new Integer(3));
108         logger.error("Hello world 4.", e);
109     }

167     private void assertLogMessage(String expected, int index) {
168         LogRecord logRecord = listHandler.recordList.get(index);
169         Assert.assertNotNull(logRecord);
170         assertEquals(expected, logRecord.getMessage());
171     }
172 
173     private void assertException(Class<? extends Throwable> exceptionType, int index) {
174         LogRecord logRecord = listHandler.recordList.get(index);
175         Assert.assertNotNull(logRecord);
176         assertEquals(exceptionType, logRecord.getThrown().getClass());
177     }

179     void removeListHandlers(java.util.logging.Logger logger) {
180         Handler[] handlers = logger.getHandlers();
181         for (Handler h : handlers) {
182             if (h instanceof ListHandler)
183                 logger.removeHandler(h);
184         }
185     }

187     static private class ListHandler extends java.util.logging.Handler {
188

198         }

202         }
